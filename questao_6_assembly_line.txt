class AssemblyLine
  def initialize(speed)
    @speed = speed
  end
  def production_rate_per_hour
    if @speed <= 4
      return (@speed * 221).to_f
    elsif @speed <=8
    return ((@speed * 221) * 0.9).to_f
    elsif @speed == 9
    return ((@speed * 221) * 0.8).to_f
    elsif @speed == 10
    return ((@speed * 221) * 0.77).to_f
    end
  end
  def working_items_per_minute
    return (production_rate_per_hour / 60).to_i
  end
end
