class LogLineParser
  def initialize(line)
    @line = line
  end

  def message
    return @line.gsub(/\[(ERROR|WARNING|INFO)*\]: / , '').strip

  end

  def log_level
    @line.scan(/\[(ERROR|WARNING|INFO)\]/).first.first.downcase
  end

  def reformat
    "#{message} (#{log_level.downcase})"
  end
end
